615 inline double MbRandom::poissonQuantile(double lambda, double p) {
616 
617  /* Starting with x = 0, find the first value for which
618  CDF(X-1) <= CDF <= CDF(X). */
619  double sum = 0.0;
620  int xmax = 100;
621  for (int i=0; i<=xmax; i++)
622  {
623  double sumOld = sum;
624  double newVal = 0.0;
625  if ( i == 0 )
626  {
627  newVal = exp(-lambda);
628  sum = newVal;
629  }
630  else
631  {
632  double last = newVal;
633  newVal = last * lambda / ( double ) ( i );
634  sum += newVal;
635  }
636  if ( sumOld <= p && p <= sum )
637  return i;
638  }

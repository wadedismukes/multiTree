376 /*
377  * Copy the elements from one vector to another, in place.
378  * That is, if you call B.inject(A), both A and B must conform
379  * (i.e. have the same dimension).
380  *
381  * This differs from B = A.copy() in that references to B
382  * before this assignment are also affected. That is, if
383  * we have
384  *
385  * MbVector A(n);
386  * MbVector C(n);
387  * MbVector B(C); (elements of B and C are shared)
388  *
389  * then B.inject(A) affects both B and C, while B=A.copy() creates
390  * a new vector B which shares no data with C or A.
391  *
392  * A is the vector from which elements will be copied.
393  * The function returns an instance of the modifed vector. That is, in
394  * B.inject(A), it returns B. If A and B are not conformant, no
395  * modifications to B are made.
396  *
397  * \brief Inject elements of A into this
398  * \param A Vector with elements to inject
399  * \return Injected vector
400  */
401 template <class T>
402 MbVector<T> &MbVector<T>::inject(const MbVector &A) {
403 
404  if (A.n == n)
405  memcpy(v, A.v, n*sizeof(T));
406  return *this;
407 
408 }
